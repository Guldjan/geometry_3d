def find_intersecting_point(line)
  eq1 = LinearEquation.new(vector.x, -line.vector.x, point.x - line.point.x)
  eq2 = LinearEquation.new(vector.y, -line.vector.y, point.y - line.point.y)
  find_point (eq1.solve_system(eq2)[0])
end
